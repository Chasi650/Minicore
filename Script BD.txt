CREATE DATABASE minicore;
Use minicore;
CREATE TABLE usuarios (
    idusuarios INT AUTO_INCREMENT PRIMARY KEY,
    nombre VARCHAR(100),
    apellido VARCHAR(100)
    -- Otros campos de usuario
);

-- Tabla de ventas
CREATE TABLE ventas (
    idventas INT AUTO_INCREMENT PRIMARY KEY,
    idusuario INT,
    fecha DATE,
    monto FLOAT
    -- Otros campos de ventas
);

-- Agrega una clave foránea para vincular las ventas con los usuarios
ALTER TABLE ventas
ADD FOREIGN KEY (idusuario)
REFERENCES usuarios (idusuarios);
